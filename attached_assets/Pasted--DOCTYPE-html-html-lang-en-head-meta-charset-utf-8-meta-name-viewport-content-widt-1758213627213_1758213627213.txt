<script>
(function(){
  const root = document.documentElement;
  const g    = document.getElementById('bars');
  const getVar = n => parseFloat(getComputedStyle(root).getPropertyValue(n));
  const midY=160, barWidth=getVar('--bar-width'), gap=getVar('--gap');
  const heights=[120,180,240,180,120];
  const count=heights.length, totalWidth=count*barWidth+(count-1)*gap, startX=-totalWidth/2;
  const bars=heights.map((h,i)=>{const x=startX+i*(barWidth+gap);const y=midY-h/2;const r=document.createElementNS('http://www.w3.org/2000/svg','rect');r.setAttribute('x',x);r.setAttribute('y',y);r.setAttribute('width',barWidth);r.setAttribute('height',h);r.setAttribute('class','bar entrance');g.appendChild(r);return r});
  const center=Math.floor(count/2), baseDelay=90;
  bars.forEach((r,i)=>{const d=Math.abs(i-center);r.style.animationDelay=(d*baseDelay)+'ms'});
  function setState(name){document.body.classList.remove('state--dormant','state--connecting','state--speaking','state--listening');document.body.classList.add('state--'+name);document.querySelectorAll('.chip').forEach(b=>{b.setAttribute('aria-pressed',b.dataset.state===name?'true':'false')});if(name!=='dormant'){bars.forEach(r=>r.classList.remove('entrance'))};const stagger=parseFloat(getComputedStyle(root).getPropertyValue('--stagger'))||120;if(name==='connecting'||name==='speaking'){bars.forEach((r,i)=>{const d=Math.abs(i-center);r.style.animationDelay=(d*stagger)+'ms'})} else {bars.forEach(r=>r.style.animationDelay='0ms')}}
  setState('dormant');
  document.querySelectorAll('.chip').forEach(btn=>{btn.addEventListener('click',()=>setState(btn.dataset.state))});
  window.addEventListener('keydown',(e)=>{if(['INPUT','TEXTAREA'].includes(document.activeElement.tagName)) return;if(e.key==='1') setState('dormant');if(e.key==='2') setState('connecting');if(e.key==='3') setState('speaking');if(e.key==='4') setState('listening')});
})();
</script>
